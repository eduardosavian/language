// Simplex Sample Program

// Variable Declaration
int a = 0, b = 0, result = 0;
float pi = 3.14;
char grade = 'A';

// Input
read a;
read b;

// Conditional Statement
if (a > b) {
    write "a is greater than b";
} else {
    write "b is greater than or equal to a";
}

// Loop (Pre-tested)
int i = 0;
while (i < 5) {
    write i;
    i = i + 1;
}

// Loop (Pre-tested with initialization and post-operation)
for (int j = 0; j < 5; j = j + 1) {
    write j;
}

// Loop (Post-tested)
int k = 0;
do {
    write k;
    k = k + 1;
} while (k < 5);

// Function Definition
proc void printMessage(string message) {
    write message;
}

// Function Call
printMessage("Hello, World!");

// Assignment
result = a + b;

// Output
write "The sum of a and b is: ";
write result;
